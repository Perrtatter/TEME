AES : 
	ENCRYPT : openssl enc -aes-256-cbc -in message.txt -out message_chiffre_aes.bin
	DECRYPT : openssl enc -d -aes-256-cbc -in message_chiffre_aes.bin -out message_dechiffre.txt 
	
RSA:
	PRIVATE : openssl genpkey -algorithm RSA -out cle_privee.pem  
	PULIC   : openssl rsa -pubout -in cle_privee.pem -out cle_publique.pem 
	ENCRYPT : openssl pkeyutl -encrypt -pubin -inkey cle_publique.pem -in message.txt -out message_chiffre_rsa.bin 
	DECRYPT : openssl pkeyutl -decrypt -inkey cle_privee.pem -in message_chiffre_rsa.bin -out message_dechiffre.txt
